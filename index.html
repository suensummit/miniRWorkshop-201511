temp <- proc.time() ## current time record
p <- 0
for (t in 0:100000){
  p <- p + 1/2*(1/4)^t
}
(spttime <- proc.time() - temp)

##   0.110   0.002   0.113
p
## [1] 0.6666667

R 語言的強項：向量運算

# temp <- proc.time()   ## current time record
# p <- 1/2
# t <- c(0:1000)
# p <- p*sum((1/4)^t)
temp <- proc.time() ## current time record
p <- 1/2*sum((1/4)^c(0:100000)) # calculate in one line
(spttime <- proc.time() - temp)
##    user  system elapsed 
##   0.004   0.000   0.004
p
## [1] 0.6666667
